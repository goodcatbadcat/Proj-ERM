# this is a check to see if a major has at least 50 factories
# which means it's not just a faction leader (which anyone can be)
is_actual_major = {
	AND = {
		is_major = yes
		OR = {
			AND = {
				num_of_factories > 49
				date < 1914.1.1
			}
			AND = {
				num_of_factories > 74
				NOT = { date < 1940.1.1 }
				date < 1915.1.1
			}
			AND = {
				num_of_factories > 99
				NOT = { date < 1942.1.1 }
				date < 1916.1.1
			}
			AND = {
				num_of_factories > 124
				NOT = { date < 1944.1.1 }
				date < 1917.1.1
			}
			num_of_factories > 149
			tag = GER
			tag = USA
			tag = ENG
			tag = AUH
			tag = RUS
			tag = ITA
			tag = FRA
			tag = JAP
		}
	}
}
